def generate_real_flag(pin):
    FLAG_PREFIX =  "CBCV{%s}"

    hashed_pin = hashlib.blake2b((pin + "blindinglights").encode("utf-8")).hexdigest()[:32]
    return FLAG_PREFIX % hashed_pin
